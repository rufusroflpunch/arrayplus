    alt_skip(num) click to toggle source

    Returns a new array with 'num' elements skipped, starting after the first element;

    [1, 2, 3, 4, 5, 6, 7].alt_skip(2) # ==> [1, 4, 7]

    # File lib/arrayplus.rb, line 21
def alt_skip(num)
  new_arr = []
  new_arr << self[0] << self.drop(1).skip(num)
  return new_arr.flatten
end

    pad(padding, num=1) click to toggle source

    Returns a new array, with new element 'padding' inserted between all elements. 'num' represents the amount for each pad.

    [1, 2, 3].pad('a') # ==> ['a', 1, 'a', 2, 'a', 3, 'a']

    # File lib/arrayplus.rb, line 62
def pad(padding, num=1)
  new_arr = []
  self.each { |i| new_arr << [padding]*num << i }
  new_arr << [padding]*num
  return new_arr.flatten
end

def pad!(padding, num=1)
  new_arr = []
  self.each { |i| new_arr << [padding]*num << i }
  new_arr << [padding]*num
  self.replace new_arr.flatten
  return self
end
    skip(num) click to toggle source

    # File lib/arrayplus.rb, line 10
def skip(num)
  new_arr = []
  self.each_slice(num + 1).to_a.each { |a| new_arr << a.drop(num) }
  return new_arr.flatten
end
